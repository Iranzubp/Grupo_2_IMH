#include <Adafruit_CCS811.h> //A単adido CCS811
//A単adido CCS811
Adafruit_CCS811 ccs;

#include <Adafruit_DHT.h>  //A単adido DHT
#include "Adafruit_CCS811.h"


// Example testing sketch for various DHT humidity/temperature sensors
// Written by ladyada, public domain

#define DHTPIN 2     // what pin we're connected to

// Uncomment whatever type you're using!
#define DHTTYPE DHT11		// DHT 11 
//#define DHTTYPE DHT22		// DHT 22 (AM2302)
//#define DHTTYPE DHT21		// DHT 21 (AM2301)


// Connect pin 1 (on the left) of the sensor to +5V
// Connect pin 2 of the sensor to whatever your DHTPIN is
// Connect pin 4 (on the right) of the sensor to GROUND
// Connect a 10K resistor from pin 2 (data) to pin 1 (power) of the sensor

#define CCS811_ADDR 0x5B //Default I2C Address
#define CCS811_ADDR 0x5A //Altenate I2C Address

double co2 = 0;

double  voc = 0;

double h = 0;
	
double t = 0;

double n = 0;

DHT dht(DHTPIN, DHTTYPE);

void setup() {
	
	Serial.begin(9600); 
	pinMode(A2, INPUT);
	Serial.println("DHTxx test!");


Particle.variable("Temperatura", t);
Particle.variable("Humedad", h);
Particle.variable("CO2", co2);
Particle.variable("VOC", voc);
Particle.variable("Noise", n);

	dht.begin();
	
        delay(2000);
	

if(!ccs.begin()){
    
Serial.println("Failed to start sensor! Please check your wiring.");
while(1);
}

}

void loop() {
// Wait a few seconds between measurements.
	delay(2000);
// Reading temperature or humidity takes about 250 milliseconds!
// Sensor readings may also be up to 2 seconds 'old' (its a very slow sensor)
 h = dht.getHumidity();
// Read temperature as Celsius
 t = dht.getTempCelcius();
//Read noise as dB
 n = analogRead(A2);

	

Serial.println("CCS811 test");  
//A単adido


    bool dataAvailable = ccs.available(); //returns true if data is available to be read
	
	bool error = ccs.readData(); //returns True if an error occurs during the read
	
	delay(2000);
	
	co2 = ccs.geteCO2(); //returns eCO2 reading
	
	delay(2000);
    
    voc = ccs.getTVOC(); //return TVOC reading

  
// Check if any reads failed and exit early (to try again).
	if (isnan(h) || isnan(t)) {
		Serial.println("Failed to read from DHT sensor!");
		return;
	}

//Print with unit values
	Serial.print("Humid: "); 
	Serial.print(h);
	Serial.print("% - ");
	Serial.print("Temp: "); 
	Serial.print(t);
	Serial.print("*C - ");
	Serial.println("CO2: ");
	Serial.println(co2);
	Serial.println("ppm");
	Serial.println("TVOC: ");
	Serial.println(voc);
	Serial.println("ppm");
	Serial.println(n);
	Serial.println("noise");

//Publish all the parameters
	Serial.println(Time.timeStr());
	Particle.publish("Humedad", String(h));
	delay(1000);
	Particle.publish("Temperatura", String(t));
	delay(1000);
	Particle.publish("eCO2", String(co2));
	delay(1000);
	Particle.publish("TVOC", String(voc));
	delay(1000);
	Particle.publish("Noise", String(n));
	
	delay(60000);
}
